碰一下按鈕、按一下按鈕與離開按鈕等3個事件產生結果

<button id="btn4">我是按鈕</button>
<p id="demo3"></p>
<script>
 function a1() {
  document.getElementById("demo3").innerHTML += "歡迎光臨，請點一下按鈕。"
 }
 function a2() {
  document.getElementById("demo3").innerHTML += "謝謝您的點擊，請離開按鈕。"
 }
 function a3() {
  document.getElementById("demo3").innerHTML += "謝謝光臨！"
 }
 var x3 = document.getElementById("btn4");
 x3.addEventListener("mouseover", a1);
 x3.addEventListener("click", a2);
 x3.addEventListener("mouseout", a3);
</script>
