		function triangleType(a, b, c) {
			if (a === b && b === c) {
				return '等邊三角形';
			} else if (a === b || b === c || a === c) {
				return '等腰三角形';

			}
		}

BMI 計算機

公式：BMI = 體重(kg) / 身高(m2)


	<p>
		請輸入您的身高、體重。<br />
		身高：<input id="cm" type="number" min="1">公分(cm)，體重：<input id="kg" type="number" min="1">公斤(kg) 
		<button id="bmiBtn" onclick="bmiBtn()">計算</button><br />
		<span id="bmiResult"></span>
	</p>
	<script>
		function bmi(cm, kg) {
			let m = cm / 100;
			let bmi = kg / (m ** 2);
			return bmi.toFixed(1);
		}

		function bmiBtn() {
			let cmVal = document.getElementById('cm').value;
			let kgVal = document.getElementById('kg').value;
			let cmNum = Number(cmVal);
			let kgNum = Number(kgVal);
			let result = document.getElementById('bmiResult');
			if (cmVal === "" || kgVal === "") {
				return result.innerHTML = '請輸入完整資料！';
			} else if (cmNum <= 0 || kgNum <= 0) {
				return result.innerHTML = '請輸入大於0的數值！';
			} else {
				let bmiVal = Number(bmi(cmNum, kgNum));
				let msg = "";
				if (bmiVal < 18.5) {
					msg = `您的BMI值是${bmiVal}，過輕！`;
				} else if (bmiVal < 24) {
					msg = `您的BMI值是${bmiVal}，正常！`;
				} else if (bmiVal < 27) {
					msg = `您的BMI值是${bmiVal}，過重！`;
				} else {
					msg = `您的BMI值是${bmiVal}，肥胖！`;
				}
				result.innerHTML = msg;
			}
		}
	</script>

判斷閏年


	<p>
		閏年：能被4整除且不能被100整除，或者能被400整除。<br />
		公式：(year % 4 === 0 && year % 100 !== 0) || (year % 400 === 0)
	</p>
	請輸入年份：<input id="inputYear" type="number"> <button id="btnYear" onclick="chkLeapYear()">判斷</button><br />
	(西元年或民國年，民國年最多為499年。)
	<p id="LeapYearResult"></p>
	<script>
		function leapyear(year) {
			return (year % 4 === 0 && year % 100 !== 0) || (year % 400 === 0);
		}

		function chkLeapYear() {
			let inputYear = document.getElementById('inputYear').value;
			let year = Number(inputYear);
			let result = document.getElementById('LeapYearResult');
			if (isNaN(year) || year <= 0) {
				return result.innerHTML = '請輸入有效年份！';
			}
			//假設民國年最長為499年。
			let roc = year < 500;
			//將民國年轉換西元年。
			let roctoad = roc ? year + 1911 : year;
			//宣告leap變數，將roctoad變數的值傳回給leapyear函式，去計算是否為閏年。
			let leap = leapyear(roctoad);
			//如果輸入民國年最長到499年，就判斷輸入的民國年是否為閏年，否則判斷輸入的西元年是否為閏年。
			if (roc) {
				result.innerHTML = `民國${year}年(西元${roctoad}年)是${leap ? '閏年' : '平年'}。`;
			} else {
				result.innerHTML = `西元${year}年(民國${roctoad - 1911}年)是${leap ? '閏年' : '平年'}。`
			}
		}
	</script>
